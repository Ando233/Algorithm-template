# 并查集

```
#include <bits/stdc++.h>
using namespace std;

const int N = 2e5 + 5;
int f[N];

int find(int x){
	return x == f[x] ? x : f[x] = find(f[x]);
}

int main(){
	int n, m;
	cin >> n >> m;
	for(int i = 1; i <= n; i++){
		f[i] = i;
	}
	for(int i = 1; i <= m; i++){
		int op, x, y;
		cin >> op >> x >> y;
		if(op == 1){
			f[find(x)] = find(y);
		}
		else{
			if(find(x) == find(y)) puts("Y");
			else puts("N");
		}
	}
} 
```
